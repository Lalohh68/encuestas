const consultarFolio = async (req, res) => {
  const numeroControl = req.params.numControl;
  try {
    const result = await services.consultarFolio(String(numeroControl));
    res.status(201);
    if (result == "") {
      res.json("aun no responde la encuesta");
    } else {
      res.json(result);
    }
  } catch (error) {
    res.status(400);
    res.send(error.message);
  }
};
